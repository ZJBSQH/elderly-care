一、当前状态

❌ 旧代码（待迁移）

AIAssistantServiceImpl — 有 Feign 数据聚合，但回答是固定模板

AIController — /ai/medicine/question、/ai/medicine/today-advice

核心问题：旧代码有数据无 AI，新代码有 AI 无数据。需要把两者整合。

二、三种集成方案对比

方案 延迟 可控性 复杂度 推荐?
① 上下文预加载
Service 层先拉数据，拼入 UserMessage 再调 AI		 最低 (1轮) 最高 最低 ⭐ 推荐
② @Tool 工具调用
Feign 封装为 @Tool，AI 自行决定何时调用		 高 (多轮) 低 中 ❌
③ RetrievalAugmentor
实现 ContentRetriever，走 LangChain4j RAG 范式		 中 中 高 过度设计
为什么选方案①？
本场景数据源固定（药品清单 + 用药记录 + 提醒任务），数据量小，每次请求都需要全量上下文。
@Tool 模式会让 AI 自己决定调不调 Feign——不可控、多一轮延迟、出错难排查。
RetrievalAugmentor 是为语义检索设计的，这里是结构化数据拉取，语义不匹配。

四、编写步骤

1

创建 MedicalContextBuilder（上下文构建器）

把 AIAssistantServiceImpl.buildContextData() 提取出来，输出从 JSON 改为自然语言文本。

原因：大模型理解自然语言远好于 JSON，且 token 消耗相当。

接口定义

package com.elderlycare.ai.service;

/**
 * 医疗上下文构建器
 * <p>
 * 聚合药品服务、提醒服务的数据，构建 AI 可理解的自然语言上下文，
 * 每个数据源独立 try-catch，单点故障不影响整体。
 *
 * @author 郑
 */
public interface MedicalContextBuilder {

    /**
     * 构建并格式化上下文为自然语言文本
     *
     * @param elderId 老人ID
     * @return 格式化后的上下文文本，供 AI 理解
     */
    String buildAndFormat(Integer elderId);
}

实现类核心逻辑（复用自旧 AIAssistantServiceImpl）

package com.elderlycare.ai.service.impl;

import cn.hutool.json.JSONUtil;
import com.elderlycare.ai.feign.MedicineFeignClient;
import com.elderlycare.ai.feign.RemindFeignClient;
import com.elderlycare.ai.service.MedicalContextBuilder;
import com.elderlycare.common.core.result.Result;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Service;

import java.util.*;

/**
 * 医疗上下文构建器实现
 * <p>
 * 依次调用药品服务和提醒服务获取数据，每个 Feign 调用独立 try-catch，
 * 单个服务异常不中断整体流程，失败的数据源置为空。
 * 最终将结构化数据格式化为大模型易于理解的自然语言文本。
 *
 * @author 郑
 */
@Slf4j
@Service
@RequiredArgsConstructor
public class MedicalContextBuilderImpl implements MedicalContextBuilder {

    /** 药品服务 Feign 客户端 */
    private final MedicineFeignClient medicineFeignClient;

    /** 提醒服务 Feign 客户端 */
    private final RemindFeignClient remindFeignClient;

    @Override
    public String buildAndFormat(Integer elderId) {
        log.info("构建医疗上下文，elderId: {}", elderId);

        // 使用 LinkedHashMap 保持数据插入顺序
        Map<String, Object> contextMap = new LinkedHashMap<>();

        // 数据源1：药品清单（独立 try-catch，失败不影响其他数据源）
        try {
            Result<List<Map<String, Object>>> result =
                medicineFeignClient.getMedicinesByElderId(elderId);
            List<Map<String, Object>> medicines =
                result != null ? result.getData() : null;
            contextMap.put("medicines", medicines != null ? medicines : List.of());
            log.info("获取药品列表成功，数量: {}",
                medicines != null ? medicines.size() : 0);
        } catch (Exception e) {
            log.error("获取药品列表失败，elderId: {}", elderId, e);
            contextMap.put("medicines", List.of());
        }

        // 数据源2：今日用药记录
        try {
            Result<List<Map<String, Object>>> result =
                medicineFeignClient.getTodayRecords(elderId);
            List<Map<String, Object>> records =
                result != null ? result.getData() : null;
            contextMap.put("todayRecords", records != null ? records : List.of());
            log.info("获取今日用药记录成功，数量: {}",
                records != null ? records.size() : 0);
        } catch (Exception e) {
            log.error("获取今日用药记录失败，elderId: {}", elderId, e);
            contextMap.put("todayRecords", List.of());
        }

        // 数据源3：今日提醒任务
        try {
            Result<List<Map<String, Object>>> result =
                remindFeignClient.getTodayTasks();
            List<Map<String, Object>> tasks =
                result != null ? result.getData() : null;
            contextMap.put("todayTasks", tasks != null ? tasks : List.of());
            log.info("获取今日提醒任务成功，数量: {}",
                tasks != null ? tasks.size() : 0);
        } catch (Exception e) {
            log.error("获取今日提醒任务失败，elderId: {}", elderId, e);
            contextMap.put("todayTasks", List.of());
        }

        // 核心改进：格式化为自然语言文本而非 JSON
        return formatToNaturalLanguage(contextMap);
    }

    /**
     * 将结构化上下文格式化为自然语言文本
     * <p>
     * 大模型对自然语言的理解能力远优于 JSON，因此将数据转化为
     * 简洁的列表式文本，每条记录一行，只保留 AI 需要的核心字段。
     *
     * @param contextMap 上下文数据 Map
     * @return 自然语言格式的上下文字符串
     */
    @SuppressWarnings("unchecked")
    private String formatToNaturalLanguage(Map<String, Object> contextMap) {
        StringBuilder sb = new StringBuilder();

        // 药品清单
        List<Map<String, Object>> medicines =
            (List<Map<String, Object>>) contextMap.getOrDefault("medicines", List.of());
        sb.append("【药品清单】\n");
        if (medicines.isEmpty()) {
            sb.append("（暂无药品数据）\n");
        } else {
            for (int i = 0; i < medicines.size(); i++) {
                Map<String, Object> m = medicines.get(i);
                sb.append(String.format("%d. %s，%s，%s，提醒时间 %s\n",
                    i + 1,
                    m.getOrDefault("medicineName", "未知"),
                    m.getOrDefault("dosage", ""),
                    m.getOrDefault("frequency", ""),
                    m.getOrDefault("remindTime", "")));
            }
        }

        // 今日用药记录
        List<Map<String, Object>> records =
            (List<Map<String, Object>>) contextMap.getOrDefault("todayRecords", List.of());
        sb.append("\n【今日用药记录】\n");
        if (records.isEmpty()) {
            sb.append("（暂无今日用药记录）\n");
        } else {
            for (int i = 0; i < records.size(); i++) {
                Map<String, Object> r = records.get(i);
                sb.append(String.format("%d. %s：%s（%s）\n",
                    i + 1,
                    r.getOrDefault("medicineName", "未知"),
                    r.getOrDefault("status", "未知"),
                    r.getOrDefault("actualTime", "未知")));
            }
        }

        // 今日提醒任务
        List<Map<String, Object>> tasks =
            (List<Map<String, Object>>) contextMap.getOrDefault("todayTasks", List.of());
        sb.append("\n【今日提醒】\n");
        if (tasks.isEmpty()) {
            sb.append("（暂无提醒任务）\n");
        } else {
            for (int i = 0; i < tasks.size(); i++) {
                Map<String, Object> t = tasks.get(i);
                sb.append(String.format("%d. %s %s\n",
                    i + 1,
                    t.getOrDefault("remindTime", ""),
                    t.getOrDefault("title", "")));
            }
        }

        return sb.toString();
    }
}
关键改进：输出自然语言而非 JSON。格式化示例效果： 
【药品清单】
1. 阿司匹林 100mg，每日1次，早餐后服用，提醒时间 08:00

【今日用药记录】
1. 阿司匹林：已服（08:05）

【今日提醒】
1. 08:00 服用阿司匹林
2. 14:00 测量血压

重写 @AiService 接口——流式 + 上下文注入

替代当前的 Assistant.java，核心改动：返回 Flux<String> 触发流式调用。

package com.elderlycare.ai.service;

import dev.langchain4j.service.SystemMessage;
import dev.langchain4j.service.UserMessage;
import dev.langchain4j.service.spring.AiService;
import reactor.core.publisher.Flux;

/**
 * 智能用药助手（LangChain4j AI Service）
 * <p>
 * 基于 DashScope 通义千问大模型，结合系统内老人的药品清单、
 * 用药记录和提醒数据，提供个性化的用药指导和健康建议。
 * 返回 Flux<String> 自动触发流式调用，逐 token 输出。
 *
 * @author 郑
 */
@AiService
public interface MedicalAssistant {

    /**
     * 基于上下文数据的智能问答
     * <p>
     * 上下文数据应在调用前由 {@link MedicalContextBuilder} 构建并拼入 userMessage。
     * LangChain4j 检测到返回类型为 Flux 时自动使用 streaming-chat-model。
     *
     * @param userMessage 包含上下文数据和用户问题的完整消息
     * @return 流式 AI 回答（逐 token）
     */
    @SystemMessage(fromResource = "prompts/medical-assistant.txt")
    Flux<String> chat(@UserMessage String userMessage);
}
为什么不用 @V 注入上下文？
@V("context") 在 SystemMessage 模板中的替换依赖 LangChain4j 版本支持，不够稳定。
改为在 Controller 层将上下文拼入 userMessage，更简单、更可控。

设计 System Prompt（中文、老年护理领域）

新建文件 resources/prompts/medical-assistant.txt

你是"银龄护医"老年护理平台的智能用药管理助手。

## 角色定位
你是一名专业的老年护理药师，专门为社区老年居民及其家属提供用药指导、
服药提醒解读和健康建议。

## 工作原则
1. 所有用药建议必须基于系统提供的真实数据，不得凭空编造。
2. 如果数据不足以支持结论，诚实说明"根据现有数据无法判断"。
3. 涉及紧急症状（剧烈疼痛、呼吸困难、意识模糊等），必须在回答末尾
   明确建议"请立即就医或拨打急救电话"。
4. 老年人常多药联用，需注意药物相互作用的风险提示。
5. 回答通俗易懂，避免过于专业的医学术语；若必须使用术语，需附带解释。

## 数据说明
系统在每次对话中会提供以下信息：
- 【药品清单】：老人的用药计划（药品名称、剂量、频次、提醒时间）
- 【今日用药记录】：今日服药情况（药品名称、状态：已服/漏服/待服）
- 【今日提醒】：今日提醒任务（时间、标题）

## 回答风格
- 先简要总结当前用药情况，再回答具体问题
- 给出可操作的建议（如"建议在饭后30分钟服用"而非"注意服药时间"）
- 语气温暖但保持专业，用"您"称呼老人，用"老人"或"家人"称呼家属
- 如果今日有漏服药物，主动提醒补服注意事项
- 回答控制在300字以内，除非问题复杂

重写 Controller——SSE 流式 + 上下文注入

合并旧 AIController 的端点路径，统一到 /ai 下。

package com.elderlycare.ai.controller;

import com.elderlycare.ai.service.MedicalAssistant;
import com.elderlycare.ai.service.MedicalContextBuilder;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.http.MediaType;
import org.springframework.web.bind.annotation.*;
import reactor.core.publisher.Flux;

/**
 * AI 助手控制器
 * <p>
 * 提供用药相关的 AI 问答接口，通过 SSE 流式返回 AI 回答。
 * 每次请求先通过 {@link MedicalContextBuilder} 拉取最新上下文数据，
 * 再交由 {@link MedicalAssistant} 调用大模型生成回答。
 *
 * @author 郑
 */
@Slf4j
@RestController
@RequestMapping("/ai")
@RequiredArgsConstructor
public class AssistantController {

    private final MedicalAssistant assistant;
    private final MedicalContextBuilder contextBuilder;

    /**
     * 药品问题 AI 问答（SSE 流式）
     * <p>
     * 接收用户问题，先拉取老人最新的用药上下文数据，
     * 拼入消息后调用大模型，以 SSE 流式逐 token 返回回答。
     *
     * @param elderId  老人ID
     * @param question 用户问题
     * @return SSE 流式 AI 回答
     */
    @PostMapping(value = "/medicine/question",
                 produces = MediaType.TEXT_EVENT_STREAM_VALUE)
    public Flux<String> askMedicine(
            @RequestParam Integer elderId,
            @RequestBody String question) {
        log.info("收到药品AI问题请求，elderId: {}, question: {}", elderId, question);

        // Flux.defer 确保每次订阅都重新拉取最新上下文
        return Flux.defer(() -> {
            String context = contextBuilder.buildAndFormat(elderId);
            String fullMessage = context + "\n\n用户问题：" + question;
            return assistant.chat(fullMessage);
        });
    }

    /**
     * 今日用药建议（SSE 流式）
     * <p>
     * 基于老人今日用药数据，自动生成用药总结和健康建议。
     *
     * @param elderId 老人ID
     * @return SSE 流式 AI 建议
     */
    @GetMapping(value = "/medicine/today-advice",
                produces = MediaType.TEXT_EVENT_STREAM_VALUE)
    public Flux<String> getTodayAdvice(@RequestParam Integer elderId) {
        log.info("收到今日用药建议请求，elderId: {}", elderId);

        return Flux.defer(() -> {
            String context = contextBuilder.buildAndFormat(elderId);
            String fullMessage = context
                + "\n\n请根据以上用药数据，给出今日用药总结和健康建议。"
                + "如发现漏服情况，请说明补服注意事项。";
            return assistant.chat(fullMessage);
        });
    }

    /**
     * 异常处理：当 AI 调用失败时，返回友好的错误提示
     * <p>
     * 该方法是 Spring WebFlux 的异常处理补充，
     * 也可在 GlobalExceptionHandler 中统一处理。
     */
    private Flux<String> errorFallback(String message) {
        return Flux.just("抱歉，" + message + "，请稍后再试。");
    }
}

五、注意事项

  1. langchain4j-reactor 已引入，返回 Flux<String> 时 LangChain4j 自动使用 streaming-chat-model（yml 已配好 qwen3.6-flash）
  2. 流式与非流式共存的配置：chat-model（同步）和 streaming-chat-model（流式）已分别配置，两者互不冲突
  3. 异常处理：ContextBuilder 内部独立 try-catch 不会抛异常。若 AI 调用失败，LangChain4j 通过 Flux error 信号传播，前端收到 SSE error 即可
  4. Token 优化：formatToNaturalLanguage() 中应只保留 AI 需要的核心字段（名称、时间、状态），去掉 createTime/updateTime 等
  5. 前后端适配：旧前端期望 Flux<ServerSentEvent<String>>（带 event 类型），新方案直接返回 Flux<String>。前端需适配——但标准 SSE 的 token 流是业界主流，适配工作量很小
  6. 若必须保留 ServerSentEvent 包装，可在 Controller 层加一行 map： 
    return assistant.chat(fullMessage)
    .map(token -> ServerSentEvent.<String>builder()
        .event("message").data(token).build());
